Basic Commands:

# Check the Ubuntu version
cat /etc/os-release

Update package list
sudo apt update

# Upgrade installed packages
sudo apt upgrade

# Install a new package
sudo apt install <package-name>

# Navigate to the home directory
cd ~

# List files and directories
ls

# Create a new directory
mkdir <directory-name>

# Create a new file
touch <file-name>
